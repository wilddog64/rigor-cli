## Change 1 — NEW `bin/ai-bootstrap`

Create this file and `chmod +x` it:

```bash
#!/usr/bin/env bash
set -euo pipefail

_repo_root() {
  git rev-parse --show-toplevel 2>/dev/null
}

_usage() {
  cat <<'EOF'
Usage: ai-bootstrap [--backend-cmd CMD] [--install]

Prepare the local lint backend for use with rigor-cli.

Options:
  --backend-cmd CMD   Backend command to verify/install (default: ruff)
  --install           Install the backend when it is missing
  -h, --help          Show this help text

Environment:
  RIGOR_PYTHON_BIN    Python interpreter used to install ruff (default: python3)
EOF
}

_warn() {
  printf 'WARN: %s\n' "$*" >&2
}

_err() {
  printf 'ERROR: %s\n' "$*" >&2
  exit 1
}

_install_ruff() {
  local py_bin="${RIGOR_PYTHON_BIN:-python3}"
  local user_bin_dir
  command -v "$py_bin" >/dev/null 2>&1 || _err "Missing Python interpreter: ${py_bin}"
  user_bin_dir="$("$py_bin" -m site --user-base)/bin"
  "$py_bin" -m pip install --user ruff
  if [[ -d "$user_bin_dir" ]]; then
    case ":$PATH:" in
      *":$user_bin_dir:"*) ;;
      *) export PATH="$user_bin_dir:$PATH" ;;
    esac
  fi
}

_rigor_bin="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd)/rigor"
[[ -x "$_rigor_bin" ]] || _err "rigor not found at ${_rigor_bin} — check rigor-cli installation"

backend_cmd="ruff"
install_backend=0

while [[ $# -gt 0 ]]; do
  case "$1" in
    --backend-cmd)
      shift
      [[ $# -gt 0 ]] || _err "--backend-cmd requires a value"
      backend_cmd="$1"
      ;;
    --install)
      install_backend=1
      ;;
    -h|--help)
      _usage
      exit 0
      ;;
    *)
      _err "Unknown argument: $1"
      ;;
  esac
  shift
done

repo_root="$(_repo_root || true)"
[[ -n "$repo_root" ]] || _err "Run from inside a git repository"
cd "$repo_root"

if [[ ! -f .github/copilot-instructions.md ]]; then
  _warn "Missing .github/copilot-instructions.md; review defaults will be weaker"
fi

if command -v "$backend_cmd" >/dev/null 2>&1; then
  printf 'Backend ready: %s\n' "$backend_cmd"
  exit 0
fi

case "$backend_cmd" in
  ruff)
    if [[ "$install_backend" -eq 1 ]]; then
      _install_ruff
      command -v "$backend_cmd" >/dev/null 2>&1 || \
        _err "ruff installation did not expose the command on PATH"
      printf 'Backend installed: %s\n' "$backend_cmd"
    else
      _err "Missing backend command: ${backend_cmd}. Re-run with --install to install it."
    fi
    ;;
  *)
    _err "Missing backend command: ${backend_cmd}. No installer is defined for this backend."
    ;;
esac
```

```bash
#!/usr/bin/env bash
set -euo pipefail

_repo_root() {
  git rev-parse --show-toplevel 2>/dev/null
}

_usage() {
  cat <<'EOF'
Usage: ai-lint [rigor lint args...]

Run rigor lint with the repo-local language backend mapping.

Environment:
  RIGOR_LINT_BACKEND_CMD   Backend command for Python files (default: ruff)
  RIGOR_LINT_BACKENDS      Full backends spec passed to rigor lint; overrides
                           RIGOR_LINT_BACKEND_CMD when set (default: py:<cmd>)
EOF
}

_warn() {
  printf 'WARN: %s\n' "$*" >&2
}

_rigor_bin="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd)/rigor"
_repo_root_path="$(_repo_root || true)"
[[ -n "$_repo_root_path" ]] || {
  _warn "Not inside a git repository; falling back to current directory"
  _repo_root_path="$(pwd)"
}

cd "$_repo_root_path"

if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
  _usage
  exit 0
fi

[[ -x "$_rigor_bin" ]] || {
  printf 'ERROR: rigor not found at %s — check rigor-cli installation\n' "$_rigor_bin" >&2
  exit 1
}

backend_cmd="${RIGOR_LINT_BACKEND_CMD:-ruff}"
export RIGOR_LINT_BACKENDS="${RIGOR_LINT_BACKENDS:-py:${backend_cmd}}"

exec "$_rigor_bin" lint "$@"
```

## Change 3 — NEW `bin/ai-review`

Create this file and `chmod +x` it:

```bash
#!/usr/bin/env bash
set -euo pipefail

_repo_root() {
  git rev-parse --show-toplevel 2>/dev/null
}

_usage() {
  cat <<'EOF'
Usage: ai-review [--prompt TEXT] [--prompt-file FILE] [--model MODEL] [--fail-on-findings|--exit-code] [rigor review args...]

Run rigor review with a configurable default prompt, stdin support, and Copilot instructions.

Piped stdin is accepted as review context:
  git diff main...HEAD | bin/ai-review

Default prompt resolution order (first match wins):
  1. RIGOR_REVIEW_DEFAULT_PROMPT env var
  2. .rigor/review-prompt file in repo root
  3. Built-in generic default

Use --fail-on-findings or --exit-code when CI should return non-zero for findings.
The review output must then end with AI_REVIEW_RESULT: findings or AI_REVIEW_RESULT: no-findings.

Environment:
  RIGOR_REVIEW_DEFAULT_PROMPT   Override the default review prompt
  AI_REVIEW_PROMPT              Override the prompt for this invocation
  AI_REVIEW_MODEL               Model to pass to rigor review
  AI_REVIEW_STREAM              Streaming mode: on|off (default: on)
  AI_REVIEW_FAIL_ON_FINDINGS    Set to 1 to enable fail-on-findings mode
  AI_RIGOR_BIN                  Path to rigor binary (default: sibling rigor in same bin/)
EOF
}

_warn() {
  printf 'WARN: %s\n' "$*" >&2
}

_redact_forbidden_fragments() {
  local text="$1"
  local fragment
  local -a forbidden=(
    "shell(git push --force)"
    "shell(git push)"
    "shell(cd"
    "shell(rm"
    "shell(eval"
    "shell(sudo"
    "shell(curl"
    "shell(wget"
  )
  for fragment in "${forbidden[@]}"; do
    text="${text//"$fragment"/[redacted restricted shell fragment]}"
  done
  printf '%s' "$text"
}

_load_default_prompt() {
  local repo_root="$1"
  if [[ -n "${RIGOR_REVIEW_DEFAULT_PROMPT:-}" ]]; then
    printf '%s' "$RIGOR_REVIEW_DEFAULT_PROMPT"
    return
  fi
  local prompt_file="$repo_root/.rigor/review-prompt"
  if [[ -f "$prompt_file" ]]; then
    cat "$prompt_file"
    return
  fi
  printf 'Review this change for correctness, backward compatibility, and missing tests or docs. Prioritize behavior regressions and security risks over style-only comments.'
}

_default_rigor_bin="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd)/rigor"
prompt_file=""
model="${AI_REVIEW_MODEL:-}"
stream_mode="${AI_REVIEW_STREAM:-on}"
fail_on_findings="${AI_REVIEW_FAIL_ON_FINDINGS:-0}"
stdin_contents=""
stdin_has_data=0

repo_root="$(_repo_root || true)"
[[ -n "$repo_root" ]] || _warn "Not inside a git repository; using current directory"
repo_root="${repo_root:-$(pwd)}"
cd "$repo_root"
rigor_bin="${AI_RIGOR_BIN:-$_default_rigor_bin}"

[[ -x "$rigor_bin" ]] || {
  printf 'ERROR: rigor not found at %s — check rigor-cli installation\n' "$rigor_bin" >&2
  exit 1
}

default_prompt="$(_load_default_prompt "$repo_root")"
prompt="${AI_REVIEW_PROMPT:-$default_prompt}"

while [[ $# -gt 0 ]]; do
  case "$1" in
    --prompt)
      shift
      [[ $# -gt 0 ]] || { printf 'ERROR: --prompt requires a value\n' >&2; exit 1; }
      prompt="$1"
      ;;
    --prompt-file)
      shift
      [[ $# -gt 0 ]] || { printf 'ERROR: --prompt-file requires a value\n' >&2; exit 1; }
      prompt_file="$1"
      ;;
    --model)
      shift
      [[ $# -gt 0 ]] || { printf 'ERROR: --model requires a value\n' >&2; exit 1; }
      model="$1"
      ;;
    --fail-on-findings|--exit-code)
      fail_on_findings=1
      ;;
    -h|--help)
      _usage
      exit 0
      ;;
    --)
      shift
      break
      ;;
    *)
      break
      ;;
  esac
  shift
done

prompt="$(_redact_forbidden_fragments "$prompt")"

if [[ ! -t 0 ]]; then
  stdin_contents="$(cat)"
  if [[ -n "$stdin_contents" ]]; then
    stdin_has_data=1
    stdin_contents="$(_redact_forbidden_fragments "$stdin_contents")"
  fi
fi

if [[ -n "$prompt_file" ]]; then
  [[ -f "$prompt_file" ]] || { printf 'ERROR: prompt file not found: %s\n' "$prompt_file" >&2; exit 1; }
  prompt_file_contents="$(_redact_forbidden_fragments "$(cat "$prompt_file")")"
else
  prompt_file_contents=""
fi

review_args=()
if [[ -n "$model" ]]; then
  review_args+=(--model "$model")
fi

has_stream=0
for arg in "$@"; do
  [[ "$arg" == "--stream" ]] && { has_stream=1; break; }
done
if [[ "$has_stream" -eq 0 ]]; then
  review_args+=(--stream "$stream_mode")
fi

if [[ ! -f .github/copilot-instructions.md ]]; then
  _warn "Missing .github/copilot-instructions.md; review will fall back to Copilot defaults"
fi

review_style_prompt='Respond in GitHub PR review style. Start with a short Summary, then Findings, then Notes or Suggestions. Keep comments concise and actionable. When possible, mention file paths and line numbers. If there are no findings, say that explicitly.'

final_prompt="$review_style_prompt

Review instruction:
$prompt"

if [[ "$fail_on_findings" -eq 1 ]]; then
  final_prompt="$final_prompt

When you finish, end with exactly one final line of either AI_REVIEW_RESULT: findings if there are one or more findings, or AI_REVIEW_RESULT: no-findings if there are no findings."
fi

if [[ -n "$prompt_file_contents" ]]; then
  final_prompt="$final_prompt

Review context from --prompt-file (${prompt_file}):
$prompt_file_contents"
fi

if [[ "$stdin_has_data" -eq 1 ]]; then
  final_prompt="$final_prompt

Review context from stdin:
$stdin_contents"
fi

if [[ "$fail_on_findings" -eq 1 ]]; then
  review_output_file="$(mktemp "${TMPDIR:-/tmp}/ai-review-output.XXXXXX")"
  cleanup_review_output_file() {
    [[ -n "${review_output_file:-}" && -f "$review_output_file" ]] && rm -f "$review_output_file"
  }
  trap cleanup_review_output_file EXIT INT TERM HUP

  set +e
  "$rigor_bin" review --prompt "$final_prompt" "${review_args[@]}" "$@" >"$review_output_file" 2>&1
  review_rc=$?
  set -e

  cat "$review_output_file"
  if [[ "$review_rc" -ne 0 ]]; then
    exit "$review_rc"
  fi
  if grep -q '^AI_REVIEW_RESULT: findings$' "$review_output_file"; then
    exit 1
  fi
  if grep -q '^AI_REVIEW_RESULT: no-findings$' "$review_output_file"; then
    exit 0
  fi

  _warn "Review output did not include AI_REVIEW_RESULT marker; failing closed"
  exit 1
fi

exec "$rigor_bin" review --prompt "$final_prompt" "${review_args[@]}" "$@"
```

## Change 4 — `scripts/tests/rigor.bats`: add 7 new tests

Append these tests after the last existing test (`rigor lint: extensionless explicit file is silently skipped`):

```bash
# ── ai-bootstrap tests ────────────────────────────────────────────────────────

_setup_ai_scripts() {
  local repo_root="${BATS_TEST_DIRNAME}/../.."
  cp "$repo_root/bin/ai-bootstrap" "$BATS_TEST_TMPDIR/bin/ai-bootstrap"
  cp "$repo_root/bin/ai-lint"      "$BATS_TEST_TMPDIR/bin/ai-lint"
  cp "$repo_root/bin/ai-review"    "$BATS_TEST_TMPDIR/bin/ai-review"
  chmod +x "$BATS_TEST_TMPDIR/bin/ai-bootstrap" \
            "$BATS_TEST_TMPDIR/bin/ai-lint" \
            "$BATS_TEST_TMPDIR/bin/ai-review"
}

@test "ai-bootstrap: errors when rigor binary is missing" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  rm -f "$BATS_TEST_TMPDIR/bin/rigor"
  run "$BATS_TEST_TMPDIR/bin/ai-bootstrap"
  [ "$status" -ne 0 ]
  [[ "$output" == *"rigor not found"* ]]
}

@test "ai-bootstrap: reports ready when backend command exists" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  printf '#!/usr/bin/env bash\nexit 0\n' > "$BATS_TEST_TMPDIR/bin/ruff"
  chmod +x "$BATS_TEST_TMPDIR/bin/ruff"
  run "$BATS_TEST_TMPDIR/bin/ai-bootstrap"
  [ "$status" -eq 0 ]
  [[ "$output" == *"Backend ready: ruff"* ]]
}

@test "ai-bootstrap: errors when backend missing and --install not given" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  run env PATH="$BATS_TEST_TMPDIR/bin" "$BATS_TEST_TMPDIR/bin/ai-bootstrap" --backend-cmd definitely-missing-xyz
  [ "$status" -ne 0 ]
  [[ "$output" == *"Missing backend command"* ]]
}

# ── ai-lint tests ─────────────────────────────────────────────────────────────

@test "ai-lint: errors when rigor binary is missing" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  rm -f "$BATS_TEST_TMPDIR/bin/rigor"
  run "$BATS_TEST_TMPDIR/bin/ai-lint"
  [ "$status" -ne 0 ]
  [[ "$output" == *"rigor not found"* ]]
}

@test "ai-lint: delegates to rigor lint with RIGOR_LINT_BACKENDS set" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  printf '#!/usr/bin/env bash\nprintf "rigor-called: %%s\\n" "$*"\n' \
    > "$BATS_TEST_TMPDIR/bin/rigor" && chmod +x "$BATS_TEST_TMPDIR/bin/rigor"
  run "$BATS_TEST_TMPDIR/bin/ai-lint"
  [ "$status" -eq 0 ]
  [[ "$output" == *"rigor-called"* ]]
  [[ "$output" == *"lint"* ]]
}

# ── ai-review tests ───────────────────────────────────────────────────────────

@test "ai-review: uses .rigor/review-prompt when present" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  mkdir -p .rigor
  printf 'my-custom-prompt' > .rigor/review-prompt
  printf '#!/usr/bin/env bash\nprintf "%%s\\n" "$*"\n' \
    > "$BATS_TEST_TMPDIR/bin/rigor" && chmod +x "$BATS_TEST_TMPDIR/bin/rigor"
  run "$BATS_TEST_TMPDIR/bin/ai-review"
  [ "$status" -eq 0 ]
  [[ "$output" == *"my-custom-prompt"* ]]
}

@test "ai-review: RIGOR_REVIEW_DEFAULT_PROMPT overrides .rigor/review-prompt" {
  cd "$BATS_TEST_TMPDIR" || exit 1
  _setup_ai_scripts
  mkdir -p .rigor
  printf 'file-prompt' > .rigor/review-prompt
  printf '#!/usr/bin/env bash\nprintf "%%s\\n" "$*"\n' \
    > "$BATS_TEST_TMPDIR/bin/rigor" && chmod +x "$BATS_TEST_TMPDIR/bin/rigor"
  RIGOR_REVIEW_DEFAULT_PROMPT="env-prompt" run "$BATS_TEST_TMPDIR/bin/ai-review"
  [ "$status" -eq 0 ]
  [[ "$output" == *"env-prompt"* ]]
  [[ "$output" != *"file-prompt"* ]]
}
```
