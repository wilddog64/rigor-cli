## Context

Copilot flagged a P1 on PR #5: `local -n` (nameref) requires Bash 4.3+. On macOS,
`/bin/bash` is 3.2 — `local -n` raises `local: -n: invalid option` and fails before
executing the target command. Since `_run_command` now always routes through
`_run_command_resolve_sudo`, this is a runtime break for any caller on older Bash.

**Fix:** Replace the nameref output mechanism with a global temp variable
`_RCRS_RUNNER`. This pattern is bash 3.2 compatible and safe for single-threaded CLI use.

## Step 2 — Edit `scripts/lib/system.sh`

### 2a — Change `_run_command_resolve_sudo` signature and output

**Remove** the `local -n _rcrs_runner="$1"` line and the first parameter (the nameref name).
The function now takes 5 arguments starting with `prog` (no nameref param).
**Replace** all `_rcrs_runner=(...)` assignments with `_RCRS_RUNNER=(...)`.

New signature comment and function opening:

```bash
# _run_command_resolve_sudo <prog> <prefer_sudo> <require_sudo> <interactive_sudo> [probe_args...]
#
# Resolves the runner array and stores it in the global _RCRS_RUNNER.
# Caller must declare _RCRS_RUNNER=() before calling and read it after.
#
# Returns 127 if --require-sudo is set but sudo is unavailable.
function _run_command_resolve_sudo() {
  local prog="$1"
  local prefer_sudo="$2"
  local require_sudo="$3"
  local interactive_sudo="$4"
  shift 4
  local -a probe_args=("$@")
```

Replace every `_rcrs_runner=(...)` with `_RCRS_RUNNER=(...)`:

```bash
  local -a sudo_flags=()
  if (( interactive_sudo == 0 )); then
    sudo_flags=(-n)
  fi

  if (( require_sudo )); then
    if (( interactive_sudo )) || sudo -n true >/dev/null 2>&1; then
      _RCRS_RUNNER=(sudo "${sudo_flags[@]}" "$prog")
    else
      echo "sudo non-interactive not available" >&2
      return 127
    fi
    return 0
  fi

  if (( ${#probe_args[@]} )); then
    if "$prog" "${probe_args[@]}" >/dev/null 2>&1; then
      _RCRS_RUNNER=("$prog")
    elif (( interactive_sudo )) && sudo "${sudo_flags[@]}" "$prog" "${probe_args[@]}" >/dev/null 2>&1; then
      _RCRS_RUNNER=(sudo "${sudo_flags[@]}" "$prog")
    elif sudo -n "$prog" "${probe_args[@]}" >/dev/null 2>&1; then
      _RCRS_RUNNER=(sudo -n "$prog")
    elif (( prefer_sudo && interactive_sudo )); then
      _RCRS_RUNNER=(sudo "${sudo_flags[@]}" "$prog")
    elif (( prefer_sudo )) && sudo -n true >/dev/null 2>&1; then
      _RCRS_RUNNER=(sudo -n "$prog")
    else
      _RCRS_RUNNER=("$prog")
    fi
    return 0
  fi

  if (( prefer_sudo && interactive_sudo )); then
    _RCRS_RUNNER=(sudo "${sudo_flags[@]}" "$prog")
  elif (( prefer_sudo )) && sudo -n true >/dev/null 2>&1; then
    _RCRS_RUNNER=(sudo -n "$prog")
  else
    _RCRS_RUNNER=("$prog")
  fi
}
```

### 2b — Update `_run_command` call site

Replace the call block (both quiet and non-quiet branches) with:

```bash
  # Decide runner: user vs sudo -n vs sudo (interactive)
  _RCRS_RUNNER=()
  if (( quiet )); then
    _run_command_resolve_sudo "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}" 2>/dev/null || {
        if (( soft )); then return 127; else exit 127; fi
      }
  else
    _run_command_resolve_sudo "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}" || {
        if (( soft )); then return 127; else exit 127; fi
      }
  fi
  local -a runner=("${_RCRS_RUNNER[@]}")
  unset _RCRS_RUNNER
```
