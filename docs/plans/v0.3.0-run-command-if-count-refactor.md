## Step 2 — Refactor `scripts/lib/system.sh`

### 2a — Add `_run_command_resolve_sudo` above `_run_command`

Insert this new private helper immediately before `_run_command`:

```bash
# _run_command_resolve_sudo <runner_nameref> <prog> <prefer_sudo> <require_sudo> <interactive_sudo> [probe_args...]
#
# Resolves the runner array (plain prog, sudo -n prog, or sudo prog) and assigns
# it to the variable named by <runner_nameref>.
#
# Returns 127 if --require-sudo is set but sudo is unavailable.
function _run_command_resolve_sudo() {
  local -n _rcrs_runner="$1"
  local prog="$2"
  local prefer_sudo="$3"
  local require_sudo="$4"
  local interactive_sudo="$5"
  shift 5
  local -a probe_args=("$@")

  local -a sudo_flags=()
  if (( interactive_sudo == 0 )); then
    sudo_flags=(-n)
  fi

  if (( require_sudo )); then
    if (( interactive_sudo )) || sudo -n true >/dev/null 2>&1; then
      _rcrs_runner=(sudo "${sudo_flags[@]}" "$prog")
    else
      echo "sudo non-interactive not available" >&2
      return 127
    fi
    return 0
  fi

  if (( ${#probe_args[@]} )); then
    if "$prog" "${probe_args[@]}" >/dev/null 2>&1; then
      _rcrs_runner=("$prog")
    elif (( interactive_sudo )) && sudo "${sudo_flags[@]}" "$prog" "${probe_args[@]}" >/dev/null 2>&1; then
      _rcrs_runner=(sudo "${sudo_flags[@]}" "$prog")
    elif sudo -n "$prog" "${probe_args[@]}" >/dev/null 2>&1; then
      _rcrs_runner=(sudo -n "$prog")
    elif (( prefer_sudo && interactive_sudo )); then
      _rcrs_runner=(sudo "${sudo_flags[@]}" "$prog")
    elif (( prefer_sudo )) && sudo -n true >/dev/null 2>&1; then
      _rcrs_runner=(sudo -n "$prog")
    else
      _rcrs_runner=("$prog")
    fi
    return 0
  fi

  if (( prefer_sudo && interactive_sudo )); then
    _rcrs_runner=(sudo "${sudo_flags[@]}" "$prog")
  elif (( prefer_sudo )) && sudo -n true >/dev/null 2>&1; then
    _rcrs_runner=(sudo -n "$prog")
  else
    _rcrs_runner=("$prog")
  fi
}
```

### 2b — Simplify `_run_command` to call the helper

Replace the runner-resolution block (lines 83–122, from `# Decide runner:` through
the closing `fi`) with a call to `_run_command_resolve_sudo`:

```bash
  local -a runner=()
  if ! _run_command_resolve_sudo runner "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}"; then
    if (( soft )); then return 127; else exit 127; fi
  fi
```

The `--quiet` flag should suppress the error from `_run_command_resolve_sudo` when
`require_sudo` fails. Wrap the call:

```bash
  local -a runner=()
  local resolve_out
  if ! resolve_out=$(
    _run_command_resolve_sudo runner "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}" 2>&1
  ); then
    (( quiet )) || echo "$resolve_out" >&2
    if (( soft )); then return 127; else exit 127; fi
  fi
```

**NOTE:** Because `_run_command_resolve_sudo` uses a nameref (`local -n`), it must
be called in the same shell process — **not** in a subshell. Use the pattern above
(capture stderr only, check return code) rather than `$(...)` for the nameref
assignment. The nameref assignment happens in the current shell; only the stderr
output is captured for the quiet check.

The correct pattern is:

```bash
  local -a runner=()
  local _rcrs_err
  if ! _rcrs_err=$(_run_command_resolve_sudo runner "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}" 2>&1 >/dev/null); then
    (( quiet )) || echo "$_rcrs_err" >&2
    if (( soft )); then return 127; else exit 127; fi
  fi
```

Wait — nameref won't work across a subshell. The right approach: call directly
(no subshell), and capture the quiet flag separately:

```bash
  local -a runner=()
  if ! _run_command_resolve_sudo runner "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}"; then
    if (( soft )); then return 127; else exit 127; fi
  fi
```

And in `_run_command_resolve_sudo`, gate the `echo` on a `quiet` parameter — but
since `_run_command_resolve_sudo` doesn't have `quiet`, let `_run_command` suppress
its stderr by redirecting `_run_command_resolve_sudo`'s output. The simplest clean
approach: always print the error in `_run_command_resolve_sudo`; in `_run_command`
call it with `2>/dev/null` when `quiet=1`:

```bash
  local -a runner=()
  if (( quiet )); then
    _run_command_resolve_sudo runner "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}" 2>/dev/null || {
      if (( soft )); then return 127; else exit 127; fi
    }
  else
    _run_command_resolve_sudo runner "$prog" \
      "$prefer_sudo" "$require_sudo" "$interactive_sudo" \
      "${probe_args[@]}" || {
      if (( soft )); then return 127; else exit 127; fi
    }
  fi
```

This is still clean and keeps each function under 8 if-blocks.

## Step 3 — Add BATS tests

File: `scripts/tests/lib/system.bats` (new file)

```bash
#!/usr/bin/env bats
# shellcheck shell=bash

setup() {
  SYSTEM_LIB="${BATS_TEST_DIRNAME}/../../lib/system.sh"
  # shellcheck source=/dev/null
  source "$SYSTEM_LIB"
}

@test "_run_command_resolve_sudo: no sudo flags → plain runner" {
  local -a runner=()
  _run_command_resolve_sudo runner "echo" 0 0 0
  [ "${runner[0]}" = "echo" ]
  [ "${#runner[@]}" -eq 1 ]
}

@test "_run_command_resolve_sudo: require_sudo unavailable → returns 127" {
  # Force sudo -n to fail by using a fake sudo
  function sudo() { return 1; }
  export -f sudo
  local -a runner=()
  run _run_command_resolve_sudo runner "echo" 0 1 0
  [ "$status" -eq 127 ]
  unset -f sudo
}

@test "_run_command_resolve_sudo: probe succeeds as user → plain runner" {
  local -a runner=()
  _run_command_resolve_sudo runner "true" 1 0 0 "--version"
  [ "${runner[0]}" = "true" ]
}

@test "_run_command: missing program → exits 127" {
  run _run_command --soft -- __nonexistent_prog_xyz__
  [ "$status" -eq 127 ]
}

@test "_run_command: succeeds for simple command" {
  run _run_command -- echo hello
  [ "$status" -eq 0 ]
  [ "$output" = "hello" ]
}

@test "_run_command: --quiet suppresses error output on missing program" {
  run _run_command --quiet --soft -- __nonexistent_prog_xyz__
  [ "$status" -eq 127 ]
  [ -z "$output" ]
}
```
